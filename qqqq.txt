        function updateUrlDisplay() {
            document.getElementById('displayUrl1').textContent = localStorage.getItem('redirectUrl1') || 'Not Set';
            document.getElementById('displayUrl2').textContent = localStorage.getItem('redirectUrl2') || 'Not Set';
            document.getElementById('displayUrl3').textContent = localStorage.getItem('redirectUrl3') || 'Not Set';
        }

        updateUrlDisplay();

        document.getElementById('submitUrls').addEventListener('click', function() {
            const url1 = document.getElementById('redirectUrl1').value;
            const url2 = document.getElementById('redirectUrl2').value;
            const url3 = document.getElementById('redirectUrl3').value;

        function copyRedirectLink(urlNumber) {
            let redirectUrl;
            switch (urlNumber) {
                case 1:
                    redirectUrl = localStorage.getItem('redirectUrl1');
                    break;
                case 2:
                    redirectUrl = localStorage.getItem('redirectUrl2');
                    break;
                case 3:
                    redirectUrl = localStorage.getItem('redirectUrl3');
                    break;
                default:
                    redirectUrl = null;
            }

            if (redirectUrl) {
                const fullUrl = ${window.location.origin}/index.html?redirect=${encodeURIComponent(redirectUrl)};
                navigator.clipboard.writeText(fullUrl).then(function() {
                    alert(`Copied URL ${urlNumber} to clipboard: ${fullUrl}`);
                }, function() {
                    alert('Failed to copy URL.');
                });
            } else {
                alert(`URL ${urlNumber} is not set.`);
            }
        }
